IT_ART_TITOLO = {
    type = character
    
    # CASO 1: ZAR
	text = {
        trigger = {
			is_female = no
            any_ruling_countries = { 
			country_rank = country_rank:rank_empire
			OR = {
				has_reform = government_reform:tsardom
				culture = { has_culture_group = culture_group:russian_group }
			}	
			}
            
        }
        localization_key = "IT_ART_LO"
    }
	
	# CASO 2: RE
    text = {
        trigger = {
            any_ruling_countries = { country_rank = country_rank:rank_kingdom }
        }
        localization_key = "IT_ART_IL"
    }
	
	# CASO 3: EARL
	text = {
        trigger = {
            any_ruling_countries = { 
			country_rank = country_rank:rank_county
			NOR = { 
			country_rank = country_rank:rank_kingdom 
			country_rank = country_rank:rank_empire 
				}
			has_reform = government_reform:irish_earldom_reform	
			}
            
        }
        localization_key = "IT_ART_L"
    }
	
	# CASO 4: JARL
	text = {
        trigger = {
			is_female = no
            any_ruling_countries = { 
			country_rank = country_rank:rank_county
			OR = {
				court_language ?= language:scandinavian_language
				AND = {
					exists = culture
					exists = culture.language
					culture.language = language:scandinavian_language
					}
				}	
			}
            
        }
        localization_key = "IT_ART_LO"
    }
	
	# CASO 5: SHOGUN
    text = {
        trigger = {
            any_ruling_countries = { 
			has_reform = government_reform:shogunate
			}
        }
        localization_key = "IT_ART_LO"
    }
	
	# CASO 6: IMPERATORE (Inizia per vocale -> L')
    text = {
        trigger = {
            any_ruling_countries = { 
			country_rank = country_rank:rank_empire 
			NOR = { 
			TAG = CHI
			has_reform = government_reform:shogunate			
				}
			}
        }
        localization_key = "IT_ART_L"
    }
	
	# CASO 7: LEADER HRE (Inizia per vocale -> L')
    text = {
        trigger = {
            any_ruling_countries = { is_leader_of_international_organization = international_organization:hre }
        }
        localization_key = "IT_ART_L"
    }
	
	# CASO 8: MEMBRO ELETTORE HRE (Elettore)
    text = {
        trigger = {
            any_ruling_countries = { 
			is_elector = yes 
			NOT = { government_type = government_type:theocracy }
			}
        }
        localization_key = "IT_ART_L"
    }
	
	# CASO 9: OTTOMANI (Inizia per vocale -> L')
    text = {
        trigger = {
            any_ruling_countries = { 
			country_rank = country_rank:rank_duchy 
			OR = {
			TAG = TUR
			TAG = AYD
				}
			}
        }
        localization_key = "IT_ART_L"
    }

    # CASO 10: REGINA / DUCHESSA (Femminile)
    text = {
        trigger = {
            is_female = yes
        }
        localization_key = "IT_ART_LA"
    }

    # CASO 11: FALLBACK GENERICO (Il Duca, Il Conte)
    text = {
        trigger = { 
            always = yes 
        }
        localization_key = "IT_ART_IL"
    }
}
